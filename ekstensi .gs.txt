function doGet(e) {
  var page = e.parameter.page || 'login'; // Default ke halaman login jika tidak ada parameter

  var template;
  switch(page) {
    case 'visi-misi':
      template = 'visiMisi';
      break;
    case 'home':
      template = 'home';
      break;
    case 'sejarah': // Rute baru untuk halaman Sejarah
      template = 'sejarah';
      break;
    case 'tupoksi': // Rute baru untuk halaman Tupoksi
      template = 'tupoksi';
      break;
    case 'struktur-organisasi': // Rute baru untuk halaman Struktur Organisasi
      template = 'strukturOrganisasi';
      break;
    default:
      template = 'index'; // Halaman login
  }

  return HtmlService.createTemplateFromFile(template)
      .evaluate()
      .setTitle(getPageTitle(template))
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

// Fungsi helper untuk mendapatkan title halaman
function getPageTitle(template) {
  const titles = {
    'index': 'Login Admin',
    'visiMisi': 'Visi & Misi',
    'home': 'Homepage',
    'sejarah': 'Sejarah',
    'tupoksi': 'Tugas Pokok & Fungsi',
    'strukturOrganisasi': 'Struktur Organisasi'
  };
  return titles[template] || 'Aplikasi Admin';
}

// Fungsi login (tetap sama seperti sebelumnya)
function processLogin(username, password) {
  const ADMIN_USERNAME = "admin";
  const ADMIN_PASSWORD = "password123";

  if (username === ADMIN_USERNAME && password === ADMIN_PASSWORD) {
    return {
      success: true,
      message: "Login Berhasil! Selamat datang, Admin.",
      redirectUrl: "?page=home" // Redirect ke homepage setelah login
    };
  } else {
    return {
      success: false,
      message: "Username atau Password salah. Silakan coba lagi."
    };
  }
}

// Fungsi include untuk menyertakan konten dari file HTML lain
function include(filename) {
  return HtmlService.createHtmlOutputFromFile(filename).getContent();
}
